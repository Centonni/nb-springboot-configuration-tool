40  /**
41   * The Spring Boot Configuration implementation of CompletionItem.
42   *
43   * It utilizes an ItemMetadata and the project classpath to render the
44   * completion item and support the documentation.
45   *
46   * @author Komi Serge Innocent &lt;komi.innocent at gmail.com&gt;
47   */
48  public class DefaultSpringBootConfigurationCompletionItem implements SpringBootConfigurationCompletionItem {
49  
50      private final ItemMetadata configurationItem;
51      private final ClassPath classPath;
52      private static final ImageIcon fieldIcon = new ImageIcon(ImageUtilities.loadImage("com/centonni/springboot/configuration/tool/sb-logo.png"));
53      private static final Color fieldColor = Color.decode("0x0000B2");
54      private final int caretOffset;
55      private final int dotOffset;
56      SpringBootConfigurationType springBootConfigurationType;
57  
58      public DefaultSpringBootConfigurationCompletionItem(ItemMetadata configurationItem, ClassPath classPath, int dotOffset, int caretOffset, SpringBootConfigurationType springBootConfigurationType) {
59          this.configurationItem = configurationItem;
60          this.classPath = classPath;
61          this.dotOffset = dotOffset;
62          this.caretOffset = caretOffset;
63          this.springBootConfigurationType = springBootConfigurationType;
64      }
65  
66      public ItemMetadata getConfigurationItem() {
67          return configurationItem;
68      }
69  
70      public ClassPath getClassPath() {
71          return classPath;
72      }
73  
74      public String getText() {
75          return configurationItem.getName();
76      }
77  
78      public String getTextRight() {
79          String type = configurationItem.getType();
80  
81          if (null == type) {
82              return null;
83          }
84  
85          int lastIndexOfDot = type.lastIndexOf(".");
86          if (lastIndexOfDot > -1 && type.length() > lastIndexOfDot) {
87              type = type.substring(lastIndexOfDot + 1);
88          }
89  
90          return type;
91      }
92  
93      @Override
94      public void defaultAction(JTextComponent jtc) {
95          try {
96              StyledDocument doc = (StyledDocument) jtc.getDocument();
97              //Here we remove the characters starting at the start offset
98              //and ending at the point where the caret is currently found:
99              doc.remove(dotOffset, caretOffset - dotOffset);
100             String text = insertionText();
101             
102             String content = doc.getText(0, doc.getLength());
103             int indexOf = content.indexOf(text);
104             if (indexOf < 0) {
105                 doc.insertString(dotOffset, text, null); 
106             }else{
107                 int end=indexOf+text.length();
108                 end=doc.getParagraphElement(end).getEndOffset();
109                 jtc.setCaretPosition(end-1);
110             }
111             //This statement will close the code completion box:
112             Completion.get().hideAll();
113         } catch (BadLocationException ex) {
114             Exceptions.printStackTrace(ex);
115         }
116     }
117 
118     @Override
119     public void processKeyEvent(KeyEvent evt) {
120     }
121 
122     @Override
123     public int getPreferredWidth(Graphics graphics, Font font) {
124         return CompletionUtilities.getPreferredWidth(getText(), getTextRight(), graphics, font);
125     }
126 
127     @Override
128     public void render(Graphics g, Font defaultFont, Color defaultColor, Color backgroundColor, int width, int height, boolean selected) {
129         String leftHtmlText = getText();
130         if (configurationItem.isDeprecated()) {
131             leftHtmlText = "<s>" + leftHtmlText + "</s>";
132         }
133         CompletionUtilities.renderHtml(fieldIcon, leftHtmlText, getTextRight(), g, defaultFont, (selected ? Color.white : fieldColor), width, height, selected);
134     }
135 
136     @Override
137     public CompletionTask createDocumentationTask() {
138         return new AsyncCompletionTask(new AsyncCompletionQuery() {
139             @Override
140             protected void query(CompletionResultSet completionResultSet, Document document, int i) {
141                 completionResultSet.setDocumentation(new SpringBootConfigurationCompletionDocumentation(DefaultSpringBootConfigurationCompletionItem.this));
142                 completionResultSet.finish();
143             }
144         });
145     }
146 
147     @Override
148     public CompletionTask createToolTipTask() {
149         return new AsyncCompletionTask(new AsyncCompletionQuery() {
150             @Override
151             protected void query(CompletionResultSet completionResultSet, Document document, int i) {
152                 JToolTip toolTip = new JToolTip();
153                 toolTip.setTipText("Press Enter to insert \"" + getText() + "\"");
154                 completionResultSet.setToolTip(toolTip);
155                 completionResultSet.finish();
156             }
157         });
158     }
159 
160     @Override
161     public boolean instantSubstitution(JTextComponent component) {
162         return false;
163     }
164 
165     @Override
166     public int getSortPriority() {
167         return 0;
168     }
169 
170     @Override
171     public CharSequence getSortText() {
172         return getText();
173     }
174 
175     @Override
176     public CharSequence getInsertPrefix() {
177         return getText();
178     }
179 
180     @Override
181     public String insertionText() {
182         if (springBootConfigurationType == SpringBootConfigurationType.PROPERTIES) {
183             return getText()+"=";
184         }
185         return new YamlItemProviderUtils(getText()).getInsertionText();
186     }
187 
188 }
