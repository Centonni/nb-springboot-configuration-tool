16  package com.centonni.springboot.configuration.tool.yaml;

18  import java.io.StringWriter;
19  import java.util.Iterator;
20  import java.util.LinkedList;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.StringTokenizer;
24  import org.yaml.snakeyaml.DumperOptions;
25  import org.yaml.snakeyaml.Yaml;
26  
27  /**
28   *
29   * @author Komi Serge Innocent &lt;komi.innocent at gmail.com&gt;
30   */
31  public class YamlItemProviderUtils {
32  
33      String item;
34  
35      public YamlItemProviderUtils(String item) {
36          this.item = item;
37      }
38  
39      public String getInsertionText() {
40          
41          String value = item;
42          
43          StringTokenizer st = new StringTokenizer(value, ".");
44          List<String> data = new LinkedList<String>();
45          while (st.hasMoreTokens()) {
46              String next = st.nextToken();
47              System.out.println(next);
48              data.add(next);
49          }
50  
51          int size = data.size();
52         
53          if (size == 4) {
54              return processItem(data, 2);
55          } else if (size == 5) {
56              return processItem(data, 3);
57          }else {
58              return processItem(data, 0);
59          }
60   
61      }
62  
63      private String processItem(List<String> data, int limit) {
64  
65          Iterator<String> iterator = data.iterator();
66          int i = 1;
67          StringBuilder builder = new StringBuilder();
68  
69          builder.append("{");
70          while (iterator.hasNext()) {
71              String next = iterator.next();
72              if (i == 1) {
73                  builder.append(next)
74                         .append(":");
75                  i++;
76                  continue;
77              } else if (i == (limit + 1) && limit != 0) {
78                  builder.append("{")
79                          .append(data.get(limit))
80                          .append(".")
81                          .append(data.get(limit+1))
82                          .append(":");
83                  i++;
84                  break;
85              }
86              builder.append("{")
87                      .append(next)
88                      .append(":");
89              i++;
90          }
91          while (i > 1) {
92              builder.append("}");
93              i--;
94          }
95          DumperOptions options = new DumperOptions();
96          options.setDefaultFlowStyle(DumperOptions.FlowStyle.BLOCK);
97          options.setPrettyFlow(true);
98          options.setIndent(4);
99          Yaml yaml = new Yaml(options);
100         Map<String, String> d = (Map<String, String>) yaml.load(builder.toString());
101 
102         StringWriter writer = new StringWriter();
103         yaml.dump(d, writer);
104         String completionValue = writer.toString();
105 
106         return completionValue.replace("null", "").trim();
107     }
108 }
